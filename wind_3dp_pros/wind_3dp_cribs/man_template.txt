; Note: To output Theta BN, use '!7h!3!DB!DN!N!N'
;        '!7h' = creates the Greek letter theta
;        '!3'  = returns to English alphabet
;        '!DB' = D makes a subscript, and prints the letter B
;        '!UB' = U makes a superscript B (fontsize = 62%)
;        '!N'  = Returns to normal level
;        '!E'  = Shifts up to superscript level (fontsize = 44%)
;        '!A'  = Shifts above current division line
;        '!B'  = Shifts below current division line
;        '!R'  = Returns to a saved position
;        '!X'  = Return to the entry font
;        '!4r' = Simple Greek sigma
;        '!7r' = Complex Greek sigma
;
;    a = alpha, c = gamma, b = beta, d = delta, e = epsilon, f = zeta
;    g = eta, h = theta, i = lota, j = kappa, k = lambda, l = mu, m = nu
;    n = xi, o = omicron, p = pi, q = rho, r = sigma, s = tau, t = upsilon
;    u = phi, v = chi, w = psi, x = omega, y = infinity
;
;        '!9=' = a not equal symbol
;        '!9+' = +/- symbol
;        '!9-' = -/+ symbol
;        '!9?' = proportional symbol
;        '!9;' = blank space
;        '!9:' = equivalent (3 horizontal lines...defined as)
;        '!9m' = greek symbol for male
;        '!9M' = blank space
;        '!9n' = greek symbol for the sun
;        '!9N' = two parallel lightning bolts
;        '!9a' =  geometry angle symbol
;        '!9A' =  ~
;        '!9b' = greater than or equal to
;        '!9B' = square
;        '!9c' = Jesus fish thingy
;        '!9C' = check mark
;        '!9d' = del operator
;        '!9D' = backwards E
;        '!9q' = script q
;        '!9s' = strange squiggle (looks like greek zeta)
;        '!9S' = square-root symbol
;        '!9x' = perpendicular symbol
;        '!9X' = times operator
;        '!9#' = parallel symbol
;     '!9! !3' = |  [Note: need a space between ! !3]
;
;        '!7D' = triangle (or capital Greek letter delta)
;        '!9B' = square
;        '!7*' = * symbol
;        '!7+' = + symbol
;        '!7v' = good chi symbol
;        '!9G' = nabla (gradient)
;        '!9V' = diamond
;        '(!9GX!3E)!Dx!N = !9d!3B!Dx!N/!9d!3t'     ; makes Faraday's Law x-direction
;
;=> Drawing Vectors  (Example:  k-vector)

;

;rhs = '!8!S!Af(x!S!Ii!R ) - f(x!S!Ij!R )!R!B x!S!Ii!R - x!S!Ij!R !X' 
;XYOUTS, 0.1 + lhs_width, 0.5, rhs, CHARSIZE = 2.,WIDTH = rhs_width, /NORMAL 
;PLOTS, [0.1 + lhs_width, 0.1 + lhs_width + rhs_width],[0.51, 0.51], /NORMAL
;

;             01 04x = !
